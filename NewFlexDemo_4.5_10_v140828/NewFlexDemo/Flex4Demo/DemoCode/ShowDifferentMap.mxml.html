            import config.ServerIP;
            
            import flash.events.MouseEvent;

            import zdims.control.Marker;

            import zdims.event.*;

            import zdims.mark.IMSMark;

            private var X:Number;
            private var Y:Number;
            //标注点逻辑坐标X
            public var logicX:Number = 12727868.7690777; 
            /* 30.54587072410364，114.3363904953003 */
            //标注点逻辑坐标Y
            public var logicY:Number = 3573911.19241591;                 
            public var  mark:IMSMark;
            [Bindable]
Embed("assets/images/marks/markimg1.png")]
            public static var testIcon:Class;
            private var viewConfig:ServerIP;
            
            /**
             * 内部组件初始化
             */
            private function tile_creationCompleteHandler():void{
                viewConfig=new ServerIP();
                viewConfig.mapConfig1(this);
                this.addEventListener(ReadURLEvent.READ_COMPLETE, readURLCompelte);
            }
            
            /**
             * 获取服务地址
             */
            private function readURLCompelte(event:ReadURLEvent):void
            {
                if(tilelayer&&viewConfig.tile2dName&&viewConfig.TileURL)
                {
                    this.tilelayer.serverAddress=viewConfig.TileURL;
                    this.tilelayer.hdfName=viewConfig.tile2dName;
                }
                if(DTileLayer)
                {
                    this.DTileLayer.serverAddress=viewConfig.TileURL;
                }
                this.mapcontainer1.refresh();
                this.mapcontainer2.refresh();
            }
            
            /**
             * 页面初始化
             */
            protected function application1_creationCompleteHandler(event:FlexEvent):void
            {
                this.mapcontainer1.addEventListener(IMSMapEvent.MAPEVENT_MAPREADY,onMapReady);//地图准备就绪事件监听
            }
            
            /**
             * 地图准备就绪事件回调
             */
            private function onMapReady(e:IMSMapEvent):void
            {
                addMap();
            }
            
            //添加图片标注
            public function addImg():void
            {                
                //初始化标注的图片对象
                var img:Image = new Image();
                img.source = testIcon;
                img.toolTip = "武汉洪山广场";
                img.height = 60;
                img.width = 60;        
                // 初始化标注对象
                mark = new IMSMark(img,CoordinateType.Logic);
                // 设置标注对象的坐标
                mark.x = logicX;
                mark.y = logicY;            
                marklayer.enableMarkHiden = false;
                // 将标注添加到绘图图层中
                marklayer.addMark(mark);
                //添加标注图片的点击事件    
                img.addEventListener(MouseEvent.CLICK,show3DMap);    
            }
            //显示城市三维景观地图窗口
            public function show3DMap(e:MouseEvent):void
            {
                addMap();
            }
            
            /**
             * 添加地图
             */
            private function addMap():void
            {
                if(this.mapwindow==null)
                {
                    this.mapwindow=new MyWindow();
                    this.mapwindow.imsmap=this.mapcontainer1;
                    this.addElement(this.mapwindow);
                }  
                updatePosition();
                if(!this.mapwindow.visible)
                    this.mapwindow.visible = true;        
            }
            
            
            /**
             * 更新地图的位置
             */
            private function updatePosition():void
            {
                //更新三维景观窗口位置,更新后标注点的窗口坐标
                var coor:Point = this.mapcontainer1.logicToScreen(logicX, logicY);        
//                //设置窗口位置点的X值
                this.X = coor.x+25 ;
//                //设置窗口位置点的Y值
                this.Y = coor.y +15;
                this.mapwindow.CenterX=this.X;
                this.mapwindow.CenterY=this.Y;
                this.mapwindow.update();
            }
