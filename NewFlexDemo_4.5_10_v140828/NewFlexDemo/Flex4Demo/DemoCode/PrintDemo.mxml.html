            private var viewConfig:ServerIP;
            
            
            protected function application1_initializeHandler(event:FlexEvent):void
            {
                viewConfig=new ServerIP();
                viewConfig.mapConfig1(this);
                this.addEventListener(ReadURLEvent.READ_COMPLETE, readURLCompelte);

            //获取服务地址
            private function readURLCompelte(event:ReadURLEvent):void
            {
                if(tileLayer&&viewConfig.tile2dName&&viewConfig.TileURL)
                {
                    this.tileLayer.serverAddress=viewConfig.TileURL;
                    this.tileLayer.hdfName=viewConfig.tile2dName;
                }
                this.mapContainer.refresh();
            }

            protected function printPreviewBtn_clickHandler(event:MouseEvent):void
            {                
                var titlewin:PrintPreviewWindow=new PrintPreviewWindow();     
                var bitmapdata:BitmapData=new BitmapData(printBorder.width,printBorder.height,true,0x000000);
                bitmapdata.draw(printBorder);
                titlewin.width=printBorder.width-100;
                titlewin.height=printBorder.height-100;
                titlewin.y=50;
                titlewin.imgData=new Bitmap(bitmapdata);
                
                PopUpManager.addPopUp(titlewin,this);
                PopUpManager.centerPopUp(titlewin);
            }

             *保存打印图

            protected function savePrintImg_clickHandler(event:MouseEvent):void
            {

            }
            
            /**
             *打印
             */
            protected function printBtn_clickHandler(event:MouseEvent):void
            {
                try{
                    var job:FlexPrintJob = new FlexPrintJob();
                    job.printAsBitmap = false;
                    if(job.start()) 
                    {
                        job.addObject(printBorder, FlexPrintJobScaleType.NONE);
                        job.send();
                    }
                }
                catch(err:Error)
                {
                    Alert.show(err.message);
                }
            }
