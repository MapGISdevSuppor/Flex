            //实例化绘图对象
            public var drawBaseArrow:DrawingBase=new DrawingBase();
            private var viewConfig:ServerIP;
            private function tile_creationCompleteHandler():void{
                viewConfig=new ServerIP();
                viewConfig.mapConfig1(this);
                this.addEventListener(ReadURLEvent.READ_COMPLETE, readURLCompelte);
            }
            //获取服务地址
            private function readURLCompelte(event:ReadURLEvent):void
            {
                this.tilelayer.serverAddress=viewConfig.TileURL;
                this.mymap.refresh();
            }
            //地图加载完成后给相关的参数赋值
            public function init():void
            {    
                this.mymap.panTo(-8.7,55.4);
                //将地图容器的id赋值给选择绘制类型的页面
                this.selctDraw.mymap=this.mymap;
                //将绘图对象赋值给选择绘制类型的页面
                this.selctDraw.drawingBase=drawBaseArrow
                //将属性设置对象赋值给选择绘制类型 的页面
                this.selctDraw.SPorty=this.selectProperty;
                //将绘图对象赋值给设置属性的页面
                this.selectProperty.drawingBase=drawBaseArrow;
            }
            //显示或隐藏标绘的窗口
            private function showSelectBox():void{
                 if(this.selctDraw.visible==false){
                    this.selctDraw.visible=true;
                }else{
                    this.selctDraw.visible=false;
                }
            }
            //显示或隐藏属性设置的窗口
            private function showSelectPro():void{
                if(this.selectProperty.visible==false){
                    this.selectProperty.visible=true;
                }else{
                    this.selectProperty.visible=false
                }
            }

            //绘制图形
            private function DrawArrow(drawType:String):void{
                this.mymap.setCurOper(IMSOperType.Drag);
                //将地图容器的id赋值给绘制图形的页面
                drawingBase.mymap=this.mymap;
                //设置绘制图形的颜色值,默认值为0xFF0010

                //设置绘制图形的Alpha值

                //绘制图形，传入绘制的类型
                drawingBase.initDrawArrow(drawType);
            }

                event.currentTarget.filters=[glower];

            private function rollOutHandler(event:Event):void{
                event.currentTarget.filters=[];
            }

            import zdimsExtends.drawingArrow.DrawingBase;
            public var drawingBase:DrawingBase;
            //选中的颜色值
            public var selectColorFill:uint=0xff0000;
            //选中的alpha值
            public var selectAlphaFill:Number=0.8;
            
            private function changeColor(num:Number,event:Event=null):void{
                selectColorFill=new uint("0x"+this.fillColor.selectedColor.toString(16));
                selectAlphaFill=fillAlpha.value;
                drawingBase.FillColor=new uint("0x"+this.fillColor.selectedColor.toString(16));
                drawingBase.Fillalpha=fillAlpha.value;
                if(num==1){
                    drawingBase.reDrawByColor(new uint("0x"+this.fillColor.selectedColor.toString(16)));
                }else if(num==2){
                    drawingBase.reDrawByAlpha(fillAlpha.value);
                }
            }
            
        ]]>
    </fx:Script>
